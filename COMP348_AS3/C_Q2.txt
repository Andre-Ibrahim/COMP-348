#include <stdio.h>
void selectionSort(int [], int);
void swap(int*, int*);
void DisplayArray(int [], int);

int main(){
    int arr[5] = {1,13,5,17,11};
    DisplayArray(arr, 5);
    selectionSort(arr, 5);
    DisplayArray(arr, 5);
    
}

void selectionSort(int arr[], int n) 
{ 
    int indexOfMin; 
  
    for (int i = 0; i < n-1; i++) 
    { 
        indexOfMin = i; 
        for (int j = i+1; j < n; j++) 
          if (arr[j] < arr[indexOfMin]) 
            indexOfMin = j; 
  
        swap(&arr[indexOfMin], &arr[i]); 
    } 
}

void swap(int *xp, int *yp) 
{ 
    int temp = *xp; 
    *xp = *yp; 
    *yp = temp; 
}

void DisplayArray(int arr[], int size){
    for(int i = 0; i < size; i++){
        printf("%d\n", arr[i]);
    }
}
